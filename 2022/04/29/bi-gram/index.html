用bi-gram实现简单汉语分词程序

思路

我们首先要将目标句子分成m个词组组成的序列，然后用bi-gram计算得到该种组成的概率，进而判断这种词组组成是否可行。参考

bi-gram:
$$
p(w_1,w_2,\dots,w_m)=p(w_1)*p(w_2\mid w_1)*p(w_3\mid w_2)*\dots *p(w_m\mid w_{m-1})
$$

最大化概率2-gram分词算法

即字面意思，选择概率最大的词组组成序列作为句子的分词结果。

import codecs
import math

class bi_gram_wordcut:
    def __init__(self):
        self.word_dict = {} # 概率 1-gram
        self.word_dict_num = {} # 词频 1-gram
        self.phrase_dict_num = {} # 词频 2-gram
        self.max_word_len = 0 # 词的最长长度
        self.all_freq = 0 # 所有词的词频总和 1-gram
        word_num_path = "./model/word_dict.model"  # 基于人民日报语料29W句子的字典
        phrase_num_path = './model/trans_dict.model'
        self.init(word_num_path, phrase_num_path)

    # 加载字典
    def init(self, word_num_path, phrase_num_path):
        self.word_dict_num = self.load_model(word_num_path)
        self.all_freq = sum(self.word_dict_num.values())
        self.max_word_len = max(len(key) for key in self.word_dict_num.keys())
        for key in self.word_dict_num:
            self.word_dict[key] = math.log(self.word_dict_num[key]/self.all_freq)

        Phrase_dict = self.load_model(phrase_num_path)
        for pre_word, pick_word in Phrase_dict.items():
            for now_word, num in pick_word.items():
                phrase = pre_word+' '+now_word
                self.phrase_dict_num[phrase] = float(num)


    def load_model(self, model_path):
        f = open(model_path, 'r')
        a = f.read()
        word_dict = eval(a)
        f.close()
        return word_dict

    # 估算未出现词的概率 即平滑
    def get_unknow_word_prob(self, word):
        return math.log(1.0/(self.all_freq**len(word)))

    # 获取词的概率
    def get_word_prob(self, word):
        if word in self.word_dict.keys():
            prob = self.word_dict[word]
        else:
            prob = self.get_unknow_word_prob(word)
        return prob

    # 2-gram概率
    def get_phrase_prob(self, pre_word, now_word):
        phrase = pre_word+' '+now_word
        if phrase in self.phrase_dict_num.keys():
            phrase_prob = math.log(self.phrase_dict_num[phrase]/self.word_dict_num[pre_word])
        else:
            phrase_prob = self.get_word_prob(now_word)
        return phrase_prob

    # 寻找最佳的前驱节点
    def get_beat_pre_node(self, sentence, node, node_list):
        # 如果node比最大词长小，取的片段长度以node的长度为限
        max_seg_len = min([node, self.max_word_len])
        pre_node_list = []  # 前驱节点列表

        # 获得所有的前驱片段，并记录累加概率
        for seg_len in range(1, max_seg_len+1):
            seg_start_node = node-seg_len
            seg = sentence[seg_start_node:node] # 获取片段
            pre_node = seg_start_node
            if pre_node == 0:
                # 如果前驱片段开始节点是序列的开始节点，则概率为<S>转移到当前词的概率
                seg_prob = self.get_phrase_prob("<BEG", seg)
            else:  # 如果不是序列开始节点，按照二元概率计算
                pre_pre_node = node_list[pre_node]["pre_node"]
                pre_pre_word = sentence[pre_pre_node:pre_node]
                seg_prob = self.get_phrase_prob(pre_pre_word, seg)

            pre_node_prob_sum = node_list[pre_node]["prob_sum"]
            now_prob_sum = pre_node_prob_sum+seg_prob
            pre_node_list.append((pre_node, now_prob_sum))

        # 找到最大的候选概率值
        (best_pre_node, best_prob_sum) = max(pre_node_list, key=lambda d: d[1])
        return best_pre_node, best_prob_sum

    # 分词函数
    def cut(self, sentence):
        sentence = sentence.strip()
        # 初始化
        node_list = []  # 记录节点的最佳前驱
        init_state = {}
        init_state["pre_node"] = -1
        init_state["prob_sum"] = 0
        node_list.append(init_state)

        # 逐个节点寻找最佳前驱节点
        for node in range(1, len(sentence)+1):
            (best_pre_node, best_prob_sum) = self.get_beat_pre_node(sentence, node, node_list)
            now_node = {}
            now_node["pre_node"] = best_pre_node
            now_node["prob_sum"] = best_prob_sum
            node_list.append(now_node)

        # 获取最优切分
        best_list = []
        node = len(sentence)
        best_list.append(node)
        while True:
            pre_node = node_list[node]["pre_node"]
            if pre_node == -1:
                break
            node = pre_node
            best_list.append(node)
        best_list.reverse()

        # 分词
        words = ""
        for i in range(len(best_list)-1):
            left = best_list[i]
            right = best_list[i+1]
            words = words+'/'+sentence[left:right]

        return words


data = []  # 分词的句子
fr = codecs.open('news_tensite_xml.smarty.dat', 'r', 'gb18030')
for i in fr:
    if "<content>" in i and "<content></content>" not in i:
        s = i.lstrip('<content>').rstrip().rstrip('</content>')
        data.append(s)
fr.close()

cuter = bi_gram_wordcut()
fh = codecs.open("分词结果.txt", 'w', 'utf-8')  # 将结果保存到 分词结果.txt
for str in data:
    words = cuter.cut(str)
    fh.write(words.strip()+'\n\n\n')
    #print(words.strip())
fh.close()

基于2-gram的前向后向最大匹配算法

先计算正向切分的词组序列概率，再计算逆向切分的词组序列概率，选择概率较大的作为分词结果

import codecs
import math


class bi_gram_wordcut:
    def __init__(self):
        self.word_dict = {}  # 词语频次字典
        self.phrase_dict = {}  # 词语后面接的词的频次
        self.all_freq = 0  # 字典词出现的总数
        self.word_type = 0  # 字典词的个数
        word_num_path = "./model/word_dict.model"  # 基于人民日报语料29W句子的字典
        phrase_num_path = './model/trans_dict.model'
        self.init(word_num_path, phrase_num_path)

    # 加载字典
    def init(self, word_num_path, phrase_num_path):
        self.word_dict = self.load_model(word_num_path)
        self.phrase_dict = self.load_model(phrase_num_path)
        self.all_freq = sum(self.word_dict.values())
        self.word_type = len(self.word_dict)

    def load_model(self, model_path):
        f = open(model_path, 'r')
        a = f.read()
        word_dict = eval(a)
        f.close()
        return word_dict

    # 最大向前分词
    def max_forward(self, sentence):
        max_len = 5
        cut_list = []
        index = 0
        while index < len(sentence):
            matched = False
            for i in range(max_len, 0, -1):
                now_word = sentence[index:index+i]
                if now_word in self.word_dict.keys():
                    cut_list.append(now_word)
                    matched = True
                    break

            # 词语与字典匹配失败
            if not matched:
                i = 1
                cut_list.append(sentence[index])
            index += i
        return cut_list

    # 最大向后分词
    def max_backforward(self, sentence):
        max_len = 5
        cut_list = []
        index = len(sentence)
        while index > 0:
            matched = False
            for i in range(max_len, 0 ,-1):
                now_word = sentence[index-i-1:index]
                if now_word in self.word_dict.keys():
                    cut_list.append(now_word)
                    matched = True
                    break

            if not matched:
                i = 0
                cut_list.append(sentence[index-1])
            index -= (i+1)
        return cut_list[::-1]

    # bi-gram计算句子概率
    def get_prob(self, seg_list):
        p = 0
        for pos, words in enumerate(seg_list):
            if pos < len(seg_list)-1:
                pre_word, now_word = words, seg_list[pos+1]
                if pre_word not in self.phrase_dict.keys():
                    # 加1平滑
                    p += math.log(1.0/self.all_freq+self.word_type)
                else:
                    # 加1平滑
                    numerator, denominator = 1.0, (self.word_type+self.word_dict[pre_word])
                    for key in self.phrase_dict[pre_word]:
                        if key == now_word:
                            numerator += self.phrase_dict[pre_word][now_word]
                            break
                    p += math.log(numerator/denominator)

            # 乘以第一个词的概率
            if (p == 0 and words!='<BEG>') or (pos == 1 and seg_list[0] == '<BEG>'):
                if words in self.word_dict.keys():
                    p += math.log((float(self.word_dict[words])+1.0)/(self.word_type+self.all_freq))
                else:
                    # 加1平滑
                    p += math.log(1.0/(self.word_type+self.all_freq))
        return p

    # 分词
    def cut(self, sentence):
        seg_list1 = self.max_forward(sentence)
        seg_list2 = self.max_backforward(sentence)
        seg_list = []
        p1 = self.get_prob(seg_list1)
        p2 = self.get_prob(seg_list2)

        if(p1 > p2):
            seg_list = seg_list1[:]
        else:
            seg_list = seg_list2[:]

        return seg_list


data = []  # 分词的句子
fr = codecs.open('news_tensite_xml.smarty.dat', 'r', 'gb18030')
for i in fr:
    if "<content>" in i and "<content></content>" not in i:
        s = i.lstrip('<content>').rstrip().rstrip('</content>')
        data.append(s)
fr.close()

cuter = bi_gram_wordcut()
fh = codecs.open("分词结果.txt", 'w', 'utf-8')
for i in data:
    seg = cuter.cut(i)
    words = '/'.join(seg)
    fh.write(words.strip()+'\n\n\n')
    #print(words.strip())
fh.close()
